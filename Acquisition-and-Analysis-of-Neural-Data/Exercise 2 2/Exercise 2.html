#Define the spike train:

def spike_train (spike_times, time_range):
    
    result = np.zeros(time_range.shape[0])
    
    timeT = np.round(time_range, decimals=4)
    
    spikeT = np.where(np.isnan(spike_times), 0, spike_times)
    
    spikeT = np.round(spike_times, decimals=4)

    for tI, t in np.ndenumerate(timeT):
        
        dirac_t = np.where(np.array(t - spikeT) == 0, 1, 0)
        result[tI] = dirac_t.sum()
        
    return result

def trial_average (spike_data, num_trials, time_range):
    
    
    result = np.zeros((num_trials, time_range.shape[0]))
    
    for trial in range(num_trials):
        
        result[trial,:] = spike_train(spike_data[:,trial], time_range)
    
    return result.mean(axis = 0)

def plot_tafr(data, \
              time_range = trial_times, \
              widths = np.array([.03, .09, .2]), \
              num_trials = np.array([10,50,100])):
    
    fig, axs = plt.subplots(3,1, figsize = (20,6))
    
    colors = ['darkred', 'tomato', 'lightsalmon']
    
    plt.tight_layout()
    
    for Itrial, trial in np.ndenumerate(num_trials):
        
        for Iwidth, width in np.ndenumerate(widths):
            
            tav = trial_average(data, trial)
            result = convolution(rec_window_func, tav, width)[:time_range.shape[0]]
        
            axs[Itrial].plot(time_range, result, \
                             linewidth = width * 100, c = colors[Iwidth[0]], \
                             alpha = 1 - (width * 3), \
                             label = "width = {}".format(width))
            axs[Itrial].set_xlabel("Time in $s$")
            axs[Itrial].set_ylabel("firing rate")
            axs[Itrial].set_title("Trial averaged firing rate for first {} trials".format(num_trials[Itrial[0]]))
            axs[Itrial].legend()
               
plot_tafr(spike_data)

The plots illustrate a smoothing effect that comes from two sources. Firstly, since the trials are independent, we can see that the variance decreases as we increase the number of trials we sample from (this is essentially the law of large numbers). Secondly, as we increase the window size the firing rates in an interval approach the temporal average of the firing rates in that interval and our firing rate estimates becomes smoother. def STA(data, trial_times, stimulus = stimulus):
    
    tav = trial_average(data, data.shape[1], trial_times)
    
    result = np.correlate(tav, stimulus, "same")
